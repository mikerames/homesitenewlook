$(function() { 
    var contentToggle = 0; 

   $('#content').on('click', function() { 
       
       if (contentToggle == 0) { 
           $('#content').animate({
               marginLeft: '40%',
       width:'60%'
           })  

           contentToggle = 1; 
       }
       else if (contentToggle == 1) {
           $('#content').animate({
               marginLeft: 0,
       width:'100%'
           }) 
           contentToggle = 0; 
       }
     
   }) 
})
